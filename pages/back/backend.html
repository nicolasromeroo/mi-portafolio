                export class RolesGuard 
                  implements CanActivate {
                  
                  canActivate(context): boolean {
                    const roles = this.reflector
                      .get('roles', context.getHandler());
                    
                    const user = context
                      .switchToHttp().getRequest().user;
                    
                    return roles.includes(user.role);
                  }
                }
